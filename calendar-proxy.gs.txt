const CALENDAR_ID = 'primary';
const DEFAULT_DAYS = 10;
const MAX_DAYS = 30;
const MEMBER_RULES = {
  alex: ['alex'],
  louisa: ['louisa', 'lou'],
  jameson: ['jameson', 'jamie']
};

function doGet(e) {
  const calendar = CalendarApp.getCalendarById(CALENDAR_ID);
  if (!calendar) {
    return ContentService
      .createTextOutput(JSON.stringify({ error: 'Calendar not found.' }))
      .setMimeType(ContentService.MimeType.JSON);
  }

  var days = parseInt(e && e.parameter && e.parameter.days, 10);
  if (!days || days < 1) days = DEFAULT_DAYS;
  if (days > MAX_DAYS) days = MAX_DAYS;

  const now = new Date();
  const end = new Date(now.getTime() + days * 24 * 60 * 60 * 1000);
  const events = calendar.getEvents(now, end).map(toEventPayload).filter(function (event) {
    return event.members.length > 0;
  });

  return ContentService
    .createTextOutput(JSON.stringify({ events: events }))
    .setMimeType(ContentService.MimeType.JSON);
}

function toEventPayload(event) {
  const haystack = [event.getTitle(), event.getDescription(), event.getLocation()]
    .join(' ')
    .toLowerCase();

  return {
    title: event.getTitle(),
    start: event.getStartTime().toISOString(),
    end: event.getEndTime().toISOString(),
    members: detectMembers(haystack),
    allDay: event.isAllDayEvent()
  };
}

function detectMembers(text) {
  return Object.keys(MEMBER_RULES).filter(function (memberId) {
    return MEMBER_RULES[memberId].some(function (token) {
      return text.indexOf(token) !== -1;
    });
  });
}
